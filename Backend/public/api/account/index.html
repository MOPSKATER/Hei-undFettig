/login

    Methoden: POST

    Felder:

      email: Valide E-Mail Addresse oder Admin
      password: SHA256 in hex, kleine Buchstaben

    Rückgabefelder:

      username: Nutzername, falls vorhanden. Sonst E-Mail Addresse
      points: Gesammelte Punkte als Ganzzahl
      accessLevel: Berechtigungsstufe als Ganzzahl
      uid: NutzerID als 16 Zeichen langer String
